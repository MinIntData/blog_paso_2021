library(stringr)
library(bsselectR)

resultados <- paste0(list.files("totalespng", full.names = TRUE))
names(resultados) <- str_replace_all(resultados, 
                                      c("\\.png" = "", 
                                        "totalespng/" = ""))

bsselect(resultados, type = "img", selected = "CABA", 
         live_search = TRUE, show_tick = TRUE)
